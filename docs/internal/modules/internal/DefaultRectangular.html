iter these_help(param d: int) where storageOrder == ArrayStorageOrder.CMO {

param rd = rank - d + 1; if rd == 1 {

for i in ranges(rd) do

yield i;

} else if rd == 2 {
for i in ranges(rd) do
for j in these_help(rank-1) do

yield (j, i);

} else {
for i in ranges(rd) do
for j in these_help(d+1) do

yield ((…j), i);

}

}
